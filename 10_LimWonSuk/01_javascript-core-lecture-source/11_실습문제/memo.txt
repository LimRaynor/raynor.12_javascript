function findWord(str, word) {
    // 코드 작성

    if (word === undefined){
        return '결과를 확인할 수 없습니다.';
    }

    // str을 ()안에 주어진 값으로 분리하여 "배열" 형태로 반환해서 arr에 저장
    const arr = str.split(',');

    // arr에서 word가 있는 인덱스(번호)를 찾아서 findIndex 변수에 저장
    // (word가 arr에 존재하면 해당 인덱스, 없으면 -1 반환)

    const findIndex = arr.indexOf(word);
    // 고양이  (배열에 word가 있을 때) == 10
    // 샌드위치(배열에 word가 없을 때) == -1

    if(findIndex == -1){
        return `${word}는 배열에 존재하지 않습니다.`;
    } else if(findIndex > -1){
        return `${word}는 배열 index ${findIndex}번째에 있습니다.`;
    }
}
